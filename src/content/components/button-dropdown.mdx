<script>{`
  document.getElementById('demo-btn-dropdown').items = [
    { text: 'Edit', id: 'edit' },
    { text: 'Delete', id: 'delete' },
    { text: 'Duplicate', id: 'duplicate' },
  ];

  const variantItems = [
    { text: 'Edit', id: 'edit' },
    { text: 'Delete', id: 'delete', disabled: true },
    { text: 'Duplicate', id: 'duplicate' },
  ];
  document.getElementById('demo-bd-primary').items = variantItems;
  document.getElementById('demo-bd-normal').items = variantItems;
  document.getElementById('demo-bd-icon').items = variantItems;

  document.getElementById('demo-bd-icons').items = [
    { id: 'launch', text: 'Launch instance', iconName: 'add-plus' },
    { id: 'stop', text: 'Stop instance', iconName: 'status-stopped' },
    { id: 'terminate', text: 'Terminate instance', iconName: 'close', disabled: true },
  ];

  document.getElementById('demo-bd-groups').items = [
    { id: 'launch', text: 'Launch instance' },
    { text: 'Instance state', items: [
      { id: 'start', text: 'Start' },
      { id: 'stop', text: 'Stop' },
      { id: 'reboot', text: 'Reboot' },
      { id: 'terminate', text: 'Terminate', disabled: true },
    ]},
  ];
`}</script>

## Usage

```html
<cs-button-dropdown>Actions</cs-button-dropdown>

<script>
  const dropdown = document.querySelector('cs-button-dropdown');
  dropdown.items = [
    { text: 'Edit', id: 'edit' },
    { text: 'Delete', id: 'delete', disabled: true },
    { text: 'Duplicate', id: 'duplicate' },
  ];
</script>
```
