## Preview

<ComponentPreview label="Basic button dropdown">
  <cs-button-dropdown id="demo-btn-dropdown">Actions</cs-button-dropdown>
</ComponentPreview>

<script>{`
  const el = document.getElementById('demo-btn-dropdown');
  if (el) {
    el.items = [
      { text: 'Edit', id: 'edit' },
      { text: 'Delete', id: 'delete' },
      { text: 'Duplicate', id: 'duplicate' },
    ];
  }
`}</script>

> **Note:** Menu items are set via the `items` JavaScript property. See usage below.

<ComponentPreview label="Variants">
  <cs-button-dropdown id="demo-bd-primary" variant="primary">Primary actions</cs-button-dropdown>
  <cs-button-dropdown id="demo-bd-normal" variant="normal">Normal actions</cs-button-dropdown>
  <cs-button-dropdown id="demo-bd-icon" variant="icon" icon-name="ellipsis" aria-label="More actions"></cs-button-dropdown>
</ComponentPreview>

<script>{`
  var bdItems = [
    { text: 'Edit', id: 'edit' },
    { text: 'Delete', id: 'delete', disabled: true },
    { text: 'Duplicate', id: 'duplicate' },
  ];
  ['demo-bd-primary', 'demo-bd-normal', 'demo-bd-icon'].forEach(function(id) {
    var el = document.getElementById(id);
    if (el) el.items = bdItems;
  });
`}</script>

<ComponentPreview label="With icons">

<script>{`
  const el = document.getElementById('demo-bd-icons');

<ComponentPreview label="Nested groups">
  <cs-button-dropdown id="demo-bd-groups" text="Instance actions"></cs-button-dropdown>

<script>{`
  const el = document.getElementById('demo-bd-groups');
  if (el) {
    el.items = [
      { id: 'launch', text: 'Launch instance' },
      { text: 'Instance state', items: [
        { id: 'start', text: 'Start' },
        { id: 'stop', text: 'Stop' },
        { id: 'reboot', text: 'Reboot' },
        { id: 'terminate', text: 'Terminate', disabled: true },
      ]},
    ];
  }
`}</script>

## Usage

```html
<cs-button-dropdown>Actions</cs-button-dropdown>

<script>
  const dropdown = document.querySelector('cs-button-dropdown');
  dropdown.items = [
    { text: 'Edit', id: 'edit' },
    { text: 'Delete', id: 'delete', disabled: true },
    { text: 'Duplicate', id: 'duplicate' },
  ];
</script>
```
