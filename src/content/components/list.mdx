## Preview

<ComponentPreview label="Default">
  <cs-list id="list-default" aria-label="Files"></cs-list>
</ComponentPreview>

<script>{`
  const items = ['report-2024.pdf', 'summary.docx', 'data.csv', 'notes.txt', 'readme.md'];
  const el = document.getElementById('list-default');
  el.items = items;
  el.renderItem = (item) => ({ id: item, content: item, secondaryContent: 'Document' });

  const icons = document.getElementById('list-icons');
  icons.items = [
    { name: 'web-server-01', type: 'EC2 Instance', state: 'running' },
    { name: 'my-bucket', type: 'S3 Bucket', state: 'active' },
    { name: 'prod-db', type: 'RDS Database', state: 'available' },
  ];
  icons.renderItem = (item) => {
    const icon = document.createElement('cs-icon');
    icon.setAttribute('name', item.state === 'running' ? 'status-positive' : 'status-info');
    return { id: item.name, content: item.name, secondaryContent: item.type, icon };
  };
`}</script>

## Usage

```html
<cs-list id="my-list" aria-label="Files"></cs-list>
<script>
  const list = document.getElementById('my-list');
  list.items = ['file1.txt', 'file2.txt'];
  list.renderItem = (item) => ({ id: item, content: item });
</script>
```
